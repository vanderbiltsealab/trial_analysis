merged2 <- merge(s_v, v_s, by = 'time_stamp')
merged2 <- merged2 %>% filter(distance.x == distance.y)
v_motion <- merged2 %>% filter(motion_category.x == 'Stationary' & motion_category.y == 'Motion')
t <- ggplot(v_motion, aes(x=time.y, y=distance_m.x)) +
  geom_line(alpha = .7) + 
  xlab("") +
  geom_hline(yintercept = 1, color = 'red') +
  theme_light() +
  theme(panel.grid.major = element_blank(), panel.grid.minor = element_blank(), panel.border = element_blank(), 
        axis.line = element_line(colour = "black"))

for(i in 1:nrow(v_motion)) {
  if (v_motion$distance_m.x[i] >= 1) {
    v_motion[i,'yn'] <- 1
  }
  else{
    v_motion[i,'yn'] <- 0 
  }
}
v_motion$sum <- rollapply(v_motion$yn,3,sum, partial = TRUE, align = "right")

for(i in 1:nrow(v_motion)) {
  if ((v_motion$sum[i] == 2) && (v_motion$sum[i-1] == 3)) {
    v_motion[i,'loop'] <- 1
  }
  else{
    v_motion[i,'loop'] <- 0 
  }
}

for(i in 1:nrow(v_motion)) {
  if ((v_motion$loop[i] == 1) && (v_motion$yn[i+1] == 1)) {
    v_motion[i,'loop'] <- 0
  }
}
sum(v_motion$loop)
